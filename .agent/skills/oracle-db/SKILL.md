## 🧠 Reglas de Oro para Oracle (Node.js Driver)

1.  **Bind Variables (SEGURIDAD)**:
    -   **OBLIGATORIO**: Jamás concatenes valores.
    -   Usa `:id`, `:codigo`, etc.
    -   *Ejemplo*: `SELECT * FROM USUARIOS WHERE RUT_ANALISTA = :rut`

2.  **Manejo de Tipos de Dato**:
    -   **Booleanos**: Oracle no tiene `BOOLEAN`. Tus tablas usan **`NUMBER(1,0)`**.
        -   App (`true`) -> DB (`1`)
        -   App (`false`) -> DB (`0`)
    -   **Fechas**:
        -   Usa `TO_DATE(:fecha, 'DD/MM/YYYY')` para columnas `DATE`.
        -   Usa `TO_TIMESTAMP(:fecha, 'DD/MM/YYYY HH24:MI:SS')` para columnas `TIMESTAMP`.
    -   **IDs Autoincrementales**: Las tablas tienen `GENERATED BY DEFAULT AS IDENTITY`. **NO** incluyas la columna ID en el `INSERT` (déjala que se genere sola).

3.  **Configuración de Salida**:
    -   Al hacer consultas en Node, siempre usa `outFormat: oracledb.OUT_FORMAT_OBJECT` para recibir JSONs limpios `{ "NOMBRE": "Valor" }` en lugar de arrays.

## 🏛️ Esquema de Tablas (Diccionario de Datos)

### 1. Núcleo & Usuarios
```sql
CREATE TABLE USUARIOS (
    RUT_ANALISTA VARCHAR2(10) PRIMARY KEY, -- Formato 12345678-9
    NOMBRE_APELLIDO_ANALISTA VARCHAR2(100),
    CORREO_ANALISTA VARCHAR2(100),
    CONTRASENA_ANALISTA VARCHAR2(100), -- Hash
    ROL_ANALISTA NUMBER(1,0) DEFAULT 0 -- 0: Analista, 1: Admin/Supervisor
);

CREATE TABLE MUESTRAS_ALI (
    CODIGO_ALI NUMBER(15,0) PRIMARY KEY, -- ID Principal del sistema
    CODIGO_OTROS NUMBER(15,0),
    OBSERVACIONES_CLIENTE VARCHAR2(4000),
    OBSERVACIONES_GENERALES VARCHAR2(4000)
);
```
### 2. Flujo TPA (Trazabilidad de pesado y Analisis)
```sql
-- Reporte Principal TPA
CREATE TABLE TPA_REPORTE (
    CODIGO_ALI NUMBER(15,0) PRIMARY KEY, -- FK a MUESTRAS_ALI
    ID_LUGAR NUMBER, -- FK LUGARES_ALMACENAMIENTO
    ESTADO_ACTUAL VARCHAR2(20), -- 'BORRADOR', 'VERIFICADO', 'NO_REALIZADO'
    FECHA_CIERRE TIMESTAMP(6),
    USUARIO_CIERRE VARCHAR2(20), -- FK USUARIOS
    FECHA_ULTIMA_MODIFICACION TIMESTAMP(6), -- Trigger TRG_TPA_AUDIT actualiza esto
    USUARIO_ULTIMA_MODIFICACION VARCHAR2(20)

-- Etapa 2: Sesiones y Equipos
CREATE TABLE TPA_ETAPA2_SESION (
    ID_SESION NUMBER IDENTITY PRIMARY KEY,
    CODIGO_ALI NUMBER(15,0),
    RUT_ANALISTA VARCHAR2(20),
    FECHA_INICIO TIMESTAMP(6),
    RETIRO_PESADO NUMBER(1,0) -- Booleano
);

CREATE TABLE TPA_ETAPA2_EQUIPOS (
    ID_USO_EQUIPO NUMBER IDENTITY PRIMARY KEY,
    ID_SESION NUMBER, -- FK TPA_ETAPA2_SESION
    ID_EQUIPO NUMBER  -- FK EQUIPOS_LAB
);

CREATE TABLE TPA_ETAPA2_MATERIALES (
    ID_USO_MATERIAL NUMBER IDENTITY PRIMARY KEY,
    ID_SESION NUMBER,
    TIPO_MATERIAL VARCHAR2(20), -- 'Punta 1ml', 'Placa 150mm', etc.
    ID_PIPETA NUMBER,      -- FK MICROPIPETAS (Opcional)
    ID_INSTRUMENTO NUMBER, -- FK INSTRUMENTOS (Opcional)
    CODIGO_MANUAL VARCHAR2(100) -- Si no es de catálogo
);

-- Etapa 3: Checklist
CREATE TABLE TPA_ETAPA3_CHECKLIST (
    ID_CHECKLIST NUMBER IDENTITY PRIMARY KEY,
    CODIGO_ALI NUMBER(15,0),
    NOMBRE_ITEM VARCHAR2(100), -- Viene de MAESTRO_CHECKLIST_LIMPIEZA
    SELECCIONADO NUMBER(1,0),
    BLOQUEADO NUMBER(1,0)
);

-- Etapa 5: Siembra y Recursos
CREATE TABLE TPA_ETAPA5_SIEMBRA (
    ID_SIEMBRA NUMBER IDENTITY PRIMARY KEY,
    CODIGO_ALI NUMBER(15,0) UNIQUE,
    OBSERVACIONES_SIEMBRA VARCHAR2(4000),
    FECHA_REGISTRO TIMESTAMP(6)
);

CREATE TABLE TPA_ETAPA5_RECURSOS (
    ID_RECURSO_SIEMBRA NUMBER IDENTITY PRIMARY KEY,
    ID_SIEMBRA NUMBER, -- FK TPA_ETAPA5_SIEMBRA
    CATEGORIA_RECURSO VARCHAR2(20), -- 'MATERIAL', 'DILUYENTE', 'EQUIPO'
    ID_MATERIAL_SIEMBRA NUMBER,
    ID_DILUYENTE NUMBER,
    ID_EQUIPO NUMBER,
    SELECCIONADO NUMBER(1,0) DEFAULT 1
);

```

### 3. Flujo Ram
```sql
CREATE TABLE RAM_REPORTE (
    CODIGO_ALI NUMBER(15,0) PRIMARY KEY, -- FK MUESTRAS_ALI
    ESTADO_RAM VARCHAR2(20) DEFAULT 'EN_PROCESO',
    AGAR_PLATE_COUNT VARCHAR2(50),
    ID_EQUIPO_INCUBACION NUMBER, -- FK EQUIPOS_INCUBACION
    FECHA_INICIO_INCUBACION DATE,
    TEMPERATURA NUMBER(5,2),
    -- Resultados
    BLANCO_UFC NUMBER,
    CONTROL_UFC NUMBER,
    CONTROL_SIEMBRA_ECOLI NUMBER,
    CONTROL_BLANCO_ESTADO VARCHAR2(20), -- 'CUMPLE', 'NO CUMPLE'
    DESFAVORABLE VARCHAR2(10) -- 'SI', 'NO'
);

CREATE TABLE RAM_ETAPA3_MUESTRAS (
    ID_MUESTRA NUMBER IDENTITY PRIMARY KEY,
    CODIGO_ALI NUMBER(15,0),
    NUMERO_MUESTRA NUMBER(5,0),
    -- Diluciones
    DIL_1 NUMBER(15,5),
    DIL_2 NUMBER(15,5),
    -- Conteos
    C1 NUMBER(10,0), C2 NUMBER(10,0), C3 NUMBER(10,0), C4 NUMBER(10,0),
    RESULTADO_RAM VARCHAR2(100), -- Texto formateado ej: '1,5 x 10^3'
    SUMATORIA_COLONIAS NUMBER(10,0)
);

CREATE TABLE RAM_ETAPA3_DUPLICADO (
    CODIGO_ALI NUMBER(15,0), -- FK RAM_REPORTE
    ID_MUESTRA_ORIGINAL NUMBER, -- FK RAM_ETAPA3_MUESTRAS
    -- Diluciones Duplicado
    DIL_DUP_1 NUMBER(15,5), DIL_DUP_2 NUMBER(15,5),
    -- Conteos Duplicado
    C_DUP_1 NUMBER(10,0), C_DUP_2 NUMBER(10,0), 
    C_DUP_3 NUMBER(10,0), C_DUP_4 NUMBER(10,0),
    -- Cálculos
    RESULTADO_RAM_DUP VARCHAR2(100),
    PROMEDIO_COLONIAS_DUPLICADO NUMBER(10,0),
    SUMATORIA_COLONIAS_DUPLICADO NUMBER(10,0)
);
```

### 4. Maestros y Catálogos (Lookups)
Úsalos para poblar dropdowns en el Frontend.

MICROPIPETAS: ID_PIPETA, NOMBRE_PIPETA, CAPACIDAD

INSTRUMENTOS: ID_INSTRUMENTO, NOMBRE_INSTRUMENTO
